105 class dimension_periodicity_proxy {
106 int &dim_;
107 int &is_periodic_;
108
109 dimension_periodicity_proxy(int &dim, int &is_periodic)
110 : dim_{dim}, is_periodic_{is_periodic} {}
111
112 public:
113 template<std::size_t N>
114 [[nodiscard]] decltype(auto) get() const {
115 if constexpr (N == 0)
116 return dim_ * 1;
117 else if constexpr (N == 1)
118 return is_periodic_ == 0 ? periodicity_tag::non_periodic
119 : periodicity_tag::periodic;
120 }
121
122 dimension_periodicity_proxy &operator=(const std::tuple<int, periodicity_tag> &t) {
123 dim_ = std::get<0>(t);
124 is_periodic_ = std::get<1>(t) == periodic;
125 return *this;
126 }

128 friend dimensions;
129 };

283 cartesian_communicator &operator=(const cartesian_communicator &other) noexcept {
284 if (this != &other) {
285 if (is_valid()) {
286 int result_1;
287 MPI_Comm_compare(comm_, MPI_COMM_WORLD, &result_1);
288 int result_2;
289 MPI_Comm_compare(comm_, MPI_COMM_SELF, &result_2);
290 if (result_1 != MPI_IDENT and result_2 != MPI_IDENT)
291 MPI_Comm_free(&comm_);
292 }
293 MPI_Comm_dup(other.comm_, &comm_);
294 }
295 return *this;
296 }

302 cartesian_communicator &operator=(cartesian_communicator &&other) noexcept {
303 if (this != &other) {
304 int result_1{0}, result_2{0};
305 MPI_Comm_compare(comm_, MPI_COMM_WORLD, &result_1);
306 MPI_Comm_compare(comm_, MPI_COMM_SELF, &result_2);
307 if (result_1 != MPI_IDENT and result_2 != MPI_IDENT)
308 MPI_Comm_free(&comm_);
309 comm_ = other.comm_;
310 other.comm_ = MPI_COMM_SELF;
311 }
312 return *this;

193 coordinates(t_dimensionality);
194 MPI_Cart_get(comm_, t_dimensionality, dimensions.data(), is_periodic_as_int.data(),
195 coordinates.data());
196 periodicity_vector is_periodic(t_dimensionality);
197 for (int i = 0; i < t_dimensionality; ++i)

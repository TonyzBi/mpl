156 class dimensions {

158
159 public:
160 class dimension_periodicity_proxy {
161 int &dim_;
162 int &is_periodic_;
163
164 dimension_periodicity_proxy(int &dim, int &is_periodic)
165 : dim_{dim}, is_periodic_{is_periodic} {}
166
167 public:
168 template<std::size_t N>
169 [[nodiscard]] decltype(auto) get() const {
170 if constexpr (N == 0)
171 return dim_ * 1;
172 else if constexpr (N == 1)
173 return is_periodic_ == 0 ? periodicity_tag::non_periodic
174 : periodicity_tag::periodic;
175 }
176
177 dimension_periodicity_proxy &operator=(const std::tuple<int, periodicity_tag> &t) {
178 dim_ = std::get<int>(t);
179 is_periodic_ = std::get<periodicity_tag>(t) == periodic;
180 return *this;
181 }
182
183 bool operator==(const std::tuple<int, periodicity_tag> &t) const {
184 return dim_ == std::get<int>(t) &&
185 (static_cast<bool>(is_periodic_) ==
186 (std::get<periodicity_tag>(t) == periodicity_tag::periodic));
187 }
188
189 bool operator!=(const std::tuple<int, periodicity_tag> &t) const {
190 return not(*this == t);
191 }
192
193 friend dimensions;
194 };

425 cartesian_communicator &operator=(const cartesian_communicator &other) noexcept {
426 if (this != &other) {
427 if (is_valid()) {
428 int result_1;
429 MPI_Comm_compare(comm_, MPI_COMM_WORLD, &result_1);
430 int result_2;
431 MPI_Comm_compare(comm_, MPI_COMM_SELF, &result_2);
432 if (result_1 != MPI_IDENT and result_2 != MPI_IDENT)
433 MPI_Comm_free(&comm_);
434 }
435 MPI_Comm_dup(other.comm_, &comm_);
436 }
437 return *this;
438 }
439
444 cartesian_communicator &operator=(cartesian_communicator &&other) noexcept {
445 if (this != &other) {
446 int result_1{0}, result_2{0};
447 MPI_Comm_compare(comm_, MPI_COMM_WORLD, &result_1);
448 MPI_Comm_compare(comm_, MPI_COMM_SELF, &result_2);
449 if (result_1 != MPI_IDENT and result_2 != MPI_IDENT)
450 MPI_Comm_free(&comm_);
451 comm_ = other.comm_;
452 other.comm_ = MPI_COMM_SELF;
453 }
454 return *this;
455 }

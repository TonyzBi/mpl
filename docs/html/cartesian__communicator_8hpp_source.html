105 class dimension_periodicity_proxy {
106 int &dim_;
107 int &is_periodic_;
108
109 dimension_periodicity_proxy(int &dim, int &is_periodic)
110 : dim_{dim}, is_periodic_{is_periodic} {}
111
112 public:
113 template<std::size_t N>
114 [[nodiscard]] decltype(auto) get() const {
115 if constexpr (N == 0)
116 return dim_ * 1;
117 else if constexpr (N == 1)
118 return is_periodic_ == 0 ? periodicity_tag::non_periodic
119 : periodicity_tag::periodic;
120 }
121
122 dimension_periodicity_proxy &operator=(const std::tuple<int, periodicity_tag> &t) {
123 dim_ = std::get<0>(t);
124 is_periodic_ = std::get<1>(t) == periodic;
125 return *this;
126 }

128 friend dimensions;

293 cartesian_communicator &operator=(const cartesian_communicator &other) noexcept {
294 if (this != &other) {
295 if (is_valid()) {
296 int result_1;
297 MPI_Comm_compare(comm_, MPI_COMM_WORLD, &result_1);
298 int result_2;
299 MPI_Comm_compare(comm_, MPI_COMM_SELF, &result_2);
300 if (result_1 != MPI_IDENT and result_2 != MPI_IDENT)
301 MPI_Comm_free(&comm_);
302 }
303 MPI_Comm_dup(other.comm_, &comm_);
304 }
305 return *this;
306 }

312 cartesian_communicator &operator=(cartesian_communicator &&other) noexcept {
313 if (this != &other) {
314 int result_1{0}, result_2{0};
315 MPI_Comm_compare(comm_, MPI_COMM_WORLD, &result_1);
316 MPI_Comm_compare(comm_, MPI_COMM_SELF, &result_2);
317 if (result_1 != MPI_IDENT and result_2 != MPI_IDENT)
318 MPI_Comm_free(&comm_);
319 comm_ = other.comm_;
320 other.comm_ = MPI_COMM_SELF;
321 }
322 return *this;

80 explicit graph_communicator(const communicator &other, const edge_set &edges,
81 bool reorder = true) {
82 int nodes{0};
83 for (const auto &e : edges) {
84#if defined MPL_DEBUG
85 if (e.first < 0 or e.second < 0)
86 throw invalid_argument();
87#endif
88 nodes = std::max({nodes, e.first + 1, e.second + 1});
89 }
90 std::vector<int> edges_list, index(nodes, 0);
91 edges_list.reserve(edges.size());
92 // the following works because the edge set is ordered with respect to the pairs of
93 // edge-node numbers
94 for (const auto &e : edges) {
95 edges_list.push_back(e.second);
96 ++index[e.first];
97 }
98 std::partial_sum(index.begin(), index.end(), index.begin());
99 MPI_Graph_create(other.comm_, nodes, index.data(), edges_list.data(), reorder, &comm_);
100 }

109 graph_communicator &operator=(const graph_communicator &other) noexcept {

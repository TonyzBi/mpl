65 distributed_grid(const cartesian_communicator &C, const sizes &size)
66 : gsize_(size.size_),
67 gbegin_(dim),
68 gend_(dim),
69 size_(dim),
70 oend_(dim),
71 overlap_(size.overlap_) {
72#if defined MPL_DEBUG
73 if (C.dimensionality() != dim or gsize_.size() != dim or overlap_.size() != dim)
74 throw invalid_dim();
75#endif
76 auto c_dimensions = C.get_dimensions();
77 auto C_coord = C.coordinate();
78 size_type vol = 1;
79 for (std::size_t i = 0; i < dim; ++i) {
80 gbegin_[i] = gbegin(gsize_[i], c_dimensions.size(i), C_coord[i]);
81 gend_[i] = gend(gsize_[i], c_dimensions.size(i), C_coord[i]);
82 size_[i] = gend_[i] - gbegin_[i];
83 oend_[i] = size_[i] + 2 * overlap_[i];
84 vol *= size_[i] + 2 * overlap_[i];
85 }
86 v.resize(vol);
87 for (std::size_t i = 0; i < dim; ++i) {
88 typename subarray_layout<T>::parameter par_m_l, par_m_r, par_b_l, par_b_r;
89 for (std::size_t j = dim - 1; true; --j) {
90 if (j == i) {
91 par_m_l.add(oend_[j], overlap_[j], 0);
92 par_m_r.add(oend_[j], overlap_[j], size_[j] + overlap_[j]);
93 par_b_l.add(oend_[j], overlap_[j], overlap_[j]);
94 par_b_r.add(oend_[j], overlap_[j], size_[j]);
95 } else {
96 par_m_l.add(oend_[j], size_[j], overlap_[j]);
97 par_m_r.add(oend_[j], size_[j], overlap_[j]);
98 par_b_l.add(oend_[j], size_[j], overlap_[j]);
99 par_b_r.add(oend_[j], size_[j], overlap_[j]);
100 }
101 if (j == 0)
102 break;
103 }
104 left_mirror_layout_.push_back(subarray_layout<T>(par_m_l));
105 right_mirror_layout_.push_back(subarray_layout<T>(par_m_r));
106 left_border_layout_.push_back(subarray_layout<T>(par_b_l));
107 right_border_layout_.push_back(subarray_layout<T>(par_b_r));
108 typename subarray_layout<T>::parameter par_i;
109 for (std::size_t j = dim - 1; true; --j) {
110 par_i.add(oend_[j], size_[j], overlap_[j]);
111 if (j == 0)
112 break;
113 }
114 interior_layout_ = subarray_layout<T>(par_i);
115 }
116 }

247 local_grid(const cartesian_communicator &C, const sizes &size) : gsize_(size.size_) {
248#if defined MPL_DEBUG
249 if (C.dimensionality() != dim or gsize_.size() != dim)
250 throw invalid_dim();
251#endif
252 size_type vol = 1;
253 for (std::size_t i = 0; i < dim; ++i)
254 vol *= gsize_[i];
255 v.resize(vol);
256 auto c_dimensions = C.get_dimensions();
257 for (int i = 0, i_end = C.size(); i < i_end; ++i) {
258 auto coords = C.coordinate(i);
259 typename subarray_layout<T>::parameter par;
260 for (std::size_t j = dim - 1; true; --j) {
261 const auto begin = gbegin(gsize_[j], c_dimensions.size(j), coords[j]);
262 const auto end = gend(gsize_[j], c_dimensions.size(j), coords[j]);
263 par.add(gsize_[j], end - begin, begin);
264 if (j == 0)
265 break;
266 }
267 sub_layout_.push_back(subarray_layout<T>(par));
268 }
269 }

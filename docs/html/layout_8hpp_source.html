78 template<typename T>
79 class layout {
80 private:
81 MPI_Datatype type_;
82
83 protected:
84 explicit layout(MPI_Datatype new_type) : type_(new_type) {
85 if (type_ != MPI_DATATYPE_NULL)
86 MPI_Type_commit(&type_);
87 }
88
89 public:
91 layout() : type_(MPI_DATATYPE_NULL) {}
92
96 layout(const layout &l) : type_(MPI_DATATYPE_NULL) {
97 if (l.type_ != MPI_DATATYPE_NULL)
98 MPI_Type_dup(l.type_, &type_);
99 }
100
104 layout(layout &&l) noexcept : type_(l.type_) { l.type_ = MPI_DATATYPE_NULL; }
105
109 layout &operator=(const layout &l) {
110 if (this != &l) {
111 if (type_ != MPI_DATATYPE_NULL)
112 MPI_Type_free(&type_);
113 if (l.type_ != MPI_DATATYPE_NULL)
114 MPI_Type_dup(l.type_, &type_);
115 else
116 type_ = MPI_DATATYPE_NULL;
117 }
118 return *this;
119 }
120
124 layout &operator=(layout &&l) noexcept {
125 if (this != &l) {
126 if (type_ != MPI_DATATYPE_NULL)
127 MPI_Type_free(&type_);
128 type_ = l.type_;
129 l.type_ = MPI_DATATYPE_NULL;
130 }
131 return *this;
132 }
133
139 [[nodiscard]] ssize_t byte_extent() const {
140 MPI_Count lb_, extent_;
141 MPI_Type_get_extent_x(type_, &lb_, &extent_);
142 if (lb_ == MPI_UNDEFINED or extent_ == MPI_UNDEFINED)
143 throw invalid_datatype_bound();
144 return extent_;
145 }
146
153 [[nodiscard]] ssize_t byte_lower_bound() const {
154 MPI_Count lb_, extent_;
155 MPI_Type_get_extent_x(type_, &lb_, &extent_);
156 if (lb_ == MPI_UNDEFINED or extent_ == MPI_UNDEFINED)
157 throw invalid_datatype_bound();
158 return lb_;
159 }

167 [[nodiscard]] ssize_t byte_upper_bound() const {
168 MPI_Count lb_, extent_;
169 MPI_Type_get_extent_x(type_, &lb_, &extent_);
170 if (lb_ == MPI_UNDEFINED or extent_ == MPI_UNDEFINED)
171 throw invalid_datatype_bound();
172 return extent_ - lb_;
173 }

181 static_assert(not std::is_void_v<T>, "layout type must be for non-void type");
182 const ssize_t res{byte_extent()};

210 const ssize_t res{byte_upper_bound()};

457 template<typename T>
458 class contiguous_layout : public layout<T> {
459 using layout<T>::type_;
460
461 static MPI_Datatype build(
462 size_t count, MPI_Datatype old_type = detail::datatype_traits<T>::get_datatype()) {
463 MPI_Datatype new_type;
464 if (count <= std::numeric_limits<int>::max()) {
465 MPI_Type_contiguous(static_cast<int>(count), old_type, &new_type);
466 } else {
467 const size_t modulus = std::numeric_limits<int>::max();
468 const size_t count1 = count / modulus;
469 const size_t count0 = count - count1 * modulus;
470 MPI_Count lb, extent;
471 MPI_Type_get_extent_x(old_type, &lb, &extent);
472 MPI_Datatype type_modulus;
473 MPI_Type_contiguous(static_cast<int>(modulus), old_type, &type_modulus);
474 std::vector<int> block_lengths{static_cast<int>(count0), static_cast<int>(count1)};
475#if defined MPL_DEBUG
476 if (count0 * extent > std::numeric_limits<MPI_Aint>::max())
477 throw invalid_size();
478#endif
479 std::vector<MPI_Aint> displacements{0, static_cast<MPI_Aint>(count0 * extent)};
480 std::vector<MPI_Datatype> types{old_type, type_modulus};
481 MPI_Type_create_struct(2, block_lengths.data(), displacements.data(), types.data(),
482 &new_type);
483 }
484 return new_type;
485 }
486
487 size_t count_;
488
489 [[nodiscard]] size_t size() const { return count_; }
490
491 public:
494 explicit contiguous_layout(size_t count = 0) : layout<T>(build(count)), count_(count) {}
495
500 explicit contiguous_layout(size_t count, const contiguous_layout<T> &l)
501 : layout<T>(build(count, l.type_)), count_(l.count_ * count) {}
502
505 contiguous_layout(const contiguous_layout<T> &l) : layout<T>(l), count_(l.count_) {}
506
509 contiguous_layout(contiguous_layout &&l) noexcept
510 : layout<T>(std::move(l)), count_(l.count_) {
511 l.count_ = 0;
512 }
513
517 contiguous_layout<T> &operator=(const contiguous_layout<T> &l) {
518 if (&l != this) {
519 layout<T>::operator=(l);
520 count_ = l.count_;
521 }
522 return *this;
523 }
524
528 contiguous_layout<T> &operator=(contiguous_layout<T> &&l) noexcept {
529 if (this != &l) {
530 layout<T>::operator=(std::move(l));
531 count_ = l.count_;
532 l.count_ = 0;
533 }
534 return *this;
535 }
536
539 void swap(contiguous_layout<T> &other) {
540 std::swap(type_, other.type_);
541 std::swap(count_, other.count_);
542 }
543
544 using layout<T>::byte_extent;
545 using layout<T>::byte_lower_bound;
546 using layout<T>::byte_upper_bound;
547 using layout<T>::extent;
548 using layout<T>::lower_bound;
549 using layout<T>::upper_bound;
550
551 friend class communicator;
552 friend class contiguous_layouts<T>;
553 };

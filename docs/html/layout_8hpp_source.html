466  class contiguous_layout : public layout<T> {
467  using layout<T>::type_;
468 
469  static MPI_Datatype build(
470  size_t count, MPI_Datatype old_type = detail::datatype_traits<T>::get_datatype()) {
471  MPI_Datatype new_type;
472  if (count <= std::numeric_limits<int>::max()) {
473  MPI_Type_contiguous(static_cast<int>(count), old_type, &new_type);
474  } else {
475  const size_t modulus = std::numeric_limits<int>::max();
476  const size_t count1 = count / modulus;
477  const size_t count0 = count - count1 * modulus;
478  MPI_Count lb, extent;
479  MPI_Type_get_extent_x(old_type, &lb, &extent);
480  MPI_Datatype type_modulus;
481  MPI_Type_contiguous(static_cast<int>(modulus), old_type, &type_modulus);
482  std::vector<int> block_lengths{static_cast<int>(count0), static_cast<int>(count1)};
483 #if defined MPL_DEBUG
484  if (count0 * extent > std::numeric_limits<MPI_Aint>::max())
485  throw invalid_size();
486 #endif
487  std::vector<MPI_Aint> displacements{0, static_cast<MPI_Aint>(count0 * extent)};
488  std::vector<MPI_Datatype> types{old_type, type_modulus};
489  MPI_Type_create_struct(2, block_lengths.data(), displacements.data(), types.data(),
490  &new_type);
491  }
492  return new_type;
493  }
494 
495  size_t count_;
496 
497  [[nodiscard]] size_t size() const { return count_; }
498 
499  public:

503

509  : layout<T>(build(count, l.type_)), count_(l.count_ * count) {}

519  l.count_ = 0;
520  }
521

528  return *this;
529  }
530 
534  contiguous_layout<T> &operator=(contiguous_layout<T> &&l) noexcept {
535  if (this != &l) {
536  layout<T>::operator=(std::move(l));
537  count_ = l.count_;
538  l.count_ = 0;
539  }
540  return *this;
541  }

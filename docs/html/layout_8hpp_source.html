81 private:
82 MPI_Datatype type_;
83
84 protected:
85 explicit layout(MPI_Datatype new_type) : type_(new_type) {
86 if (type_ != MPI_DATATYPE_NULL)
87 MPI_Type_commit(&type_);
88 }
89
90 public:
92 layout() : type_(MPI_DATATYPE_NULL) {}
93
97 layout(const layout &l) : type_(MPI_DATATYPE_NULL) {
98 if (l.type_ != MPI_DATATYPE_NULL)
99 MPI_Type_dup(l.type_, &type_);
100 }
101
105 layout(layout &&l) noexcept : type_(l.type_) { l.type_ = MPI_DATATYPE_NULL; }
106
110 layout &operator=(const layout &l) {
111 if (this != &l) {
112 if (type_ != MPI_DATATYPE_NULL)
113 MPI_Type_free(&type_);
114 if (l.type_ != MPI_DATATYPE_NULL)
115 MPI_Type_dup(l.type_, &type_);
116 else
117 type_ = MPI_DATATYPE_NULL;
118 }
119 return *this;
120 }
121
125 layout &operator=(layout &&l) noexcept {
126 if (this != &l) {
127 if (type_ != MPI_DATATYPE_NULL)
128 MPI_Type_free(&type_);
129 type_ = l.type_;
130 l.type_ = MPI_DATATYPE_NULL;
131 }
132 return *this;
133 }
134
140 [[nodiscard]] ssize_t byte_extent() const {
141 MPI_Count lb_, extent_;
142 MPI_Type_get_extent_x(type_, &lb_, &extent_);
143 if (lb_ == MPI_UNDEFINED or extent_ == MPI_UNDEFINED)
144 throw invalid_datatype_bound();
145 return extent_;
146 }
147
154 [[nodiscard]] ssize_t byte_lower_bound() const {
155 MPI_Count lb_, extent_;
156 MPI_Type_get_extent_x(type_, &lb_, &extent_);

169 MPI_Count lb_, extent_;

458 template<typename T>
459 class contiguous_layout : public layout<T> {
460 using layout<T>::type_;
461
462 static MPI_Datatype build(
463 size_t count, MPI_Datatype old_type = detail::datatype_traits<T>::get_datatype()) {
464 MPI_Datatype new_type;
465 if (count <= static_cast<size_t>(std::numeric_limits<int>::max())) {
466 MPI_Type_contiguous(static_cast<int>(count), old_type, &new_type);
467 } else {
468 const size_t modulus{std::numeric_limits<int>::max()};
469 const size_t count_1{count / modulus};
470 const size_t count_0{count - count_1 * modulus};
471 MPI_Count lb, extent;
472 MPI_Type_get_extent_x(old_type, &lb, &extent);
473 MPI_Datatype type_modulus;
474 MPI_Type_contiguous(static_cast<int>(modulus), old_type, &type_modulus);
475 std::vector<int> block_lengths{static_cast<int>(count_0), static_cast<int>(count_1)};
476#if defined MPL_DEBUG
477 if (count_0 * extent > static_cast<size_t>(std::numeric_limits<MPI_Aint>::max()))
478 throw invalid_size();
479#endif
480 std::vector<MPI_Aint> displacements{0, static_cast<MPI_Aint>(count_0 * extent)};
481 std::vector<MPI_Datatype> types{old_type, type_modulus};
482 MPI_Type_create_struct(2, block_lengths.data(), displacements.data(), types.data(),
483 &new_type);
484 }
485 return new_type;
486 }
487
488 size_t count_;
489
490 [[nodiscard]] size_t size() const { return count_; }
491
492 public:
495 explicit contiguous_layout(size_t count = 0) : layout<T>(build(count)), count_(count) {}
496
501 explicit contiguous_layout(size_t count, const contiguous_layout<T> &l)
502 : layout<T>(build(count, l.type_)), count_(l.count_ * count) {}
503
506 contiguous_layout(const contiguous_layout<T> &l) : layout<T>(l), count_(l.count_) {}
507
510 contiguous_layout(contiguous_layout &&l) noexcept
511 : layout<T>(std::move(l)), count_(l.count_) {
512 l.count_ = 0;
513 }
514
518 contiguous_layout<T> &operator=(const contiguous_layout<T> &l) {
519 if (&l != this) {
520 layout<T>::operator=(l);
521 count_ = l.count_;
522 }
523 return *this;
524 }
525
529 contiguous_layout<T> &operator=(contiguous_layout<T> &&l) noexcept {
530 if (this != &l) {
531 layout<T>::operator=(std::move(l));
532 count_ = l.count_;
533 l.count_ = 0;
534 }
535 return *this;
536 }
537
540 void swap(contiguous_layout<T> &other) {
541 std::swap(type_, other.type_);
542 std::swap(count_, other.count_);
543 }
544
545 using layout<T>::byte_extent;
546 using layout<T>::byte_lower_bound;
547 using layout<T>::byte_upper_bound;
548 using layout<T>::extent;
549 using layout<T>::lower_bound;
550 using layout<T>::upper_bound;
551
552 friend class impl::base_communicator;
553 friend class communicator;
554 friend class contiguous_layouts<T>;
555 };

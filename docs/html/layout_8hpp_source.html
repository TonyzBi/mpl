458  class contiguous_layout : public layout<T> {
459  using layout<T>::type_;
460 
461  static MPI_Datatype build(
462  size_t count, MPI_Datatype old_type = detail::datatype_traits<T>::get_datatype()) {
463  MPI_Datatype new_type;
464  if (count <= std::numeric_limits<int>::max()) {
465  MPI_Type_contiguous(static_cast<int>(count), old_type, &new_type);
466  } else {
467  const size_t modulus = std::numeric_limits<int>::max();
468  const size_t count1 = count / modulus;
469  const size_t count0 = count - count1 * modulus;
470  MPI_Count lb, extent;
471  MPI_Type_get_extent_x(old_type, &lb, &extent);
472  MPI_Datatype type_modulus;
473  MPI_Type_contiguous(static_cast<int>(modulus), old_type, &type_modulus);
474  std::vector<int> block_lengths{static_cast<int>(count0), static_cast<int>(count1)};
475 #if defined MPL_DEBUG
476  if (count0 * extent > std::numeric_limits<MPI_Aint>::max())
477  throw invalid_size();
478 #endif
479  std::vector<MPI_Aint> displacements{0, static_cast<MPI_Aint>(count0 * extent)};
480  std::vector<MPI_Datatype> types{old_type, type_modulus};
481  MPI_Type_create_struct(2, block_lengths.data(), displacements.data(), types.data(),
482  &new_type);
483  }
484  return new_type;
485  }
486 
487  size_t count_;
488 
489  [[nodiscard]] size_t size() const { return count_; }
490 
491  public:
494  explicit contiguous_layout(size_t count = 0) : layout<T>(build(count)), count_(count) {}
495 
500  explicit contiguous_layout(size_t count, const contiguous_layout<T> &l)
501  : layout<T>(build(count, l.type_)), count_(l.count_ * count) {}
502 
505  contiguous_layout(const contiguous_layout<T> &l) : layout<T>(l), count_(l.count_) {}
506 
509  contiguous_layout(contiguous_layout &&l) noexcept
510  : layout<T>(std::move(l)), count_(l.count_) {
511  l.count_ = 0;
512  }
513 
517  contiguous_layout<T> &operator=(const contiguous_layout<T> &l) {
518  if (&l != this) {

520  count_ = l.count_;
521  }

523  }
524 
528  contiguous_layout<T> &operator=(contiguous_layout<T> &&l) noexcept {
529  if (this != &l) {
530  layout<T>::operator=(std::move(l));
531  count_ = l.count_;
532  l.count_ = 0;
533  }
534  return *this;
535  }

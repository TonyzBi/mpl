307 template<typename T>
308 void neighbor_alltoallv(const T *senddata, const layouts<T> &sendls,
309 const displacements &senddispls, T *recvdata,
310 const layouts<T> &recvls, const displacements &recvdispls) const {
311 const std::vector<int> counts(recvls.size(), 1);
312 static_assert(
313 sizeof(decltype(*sendls())) == sizeof(MPI_Datatype),
314 "compiler adds some unexpected padding, reinterpret cast will yield wrong results");
315 MPI_Neighbor_alltoallw(senddata, counts.data(), senddispls(),
316 reinterpret_cast<const MPI_Datatype *>(sendls()), recvdata,
317 counts.data(), recvdispls(),
318 reinterpret_cast<const MPI_Datatype *>(recvls()), comm_);
319 }

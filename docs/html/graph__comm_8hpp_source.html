51 explicit graph_communicator(const communicator &old_comm, const edge_set &es,
52 bool reorder = true) {
53 int nodes = 0;
54 for (const auto &e : es) {
55#if defined MPL_DEBUG
56 if (e.first < 0 or e.second < 0)
57 throw invalid_argument();
58#endif
59 nodes = std::max({nodes, e.first + 1, e.second + 1});
60 }
61 std::vector<int> edges, index(nodes, 0);
62 edges.reserve(es.size());
63 // the following works because the edge set is ordered with respect to the pairs of
64 // edge-node numbers
65 for (const auto &e : es) {
66 edges.push_back(e.second);
67 ++index[e.first];
68 }
69 std::partial_sum(index.begin(), index.end(), index.begin());
70 MPI_Graph_create(old_comm.comm_, nodes, index.data(), edges.data(), reorder, &comm_);
71 }

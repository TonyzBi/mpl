226 template<typename T, typename F>
227 class op {
228 public:
229 using functor = F;
230
231 static_assert(is_binary_functor<T, F>::value,
232 "reduction operator must be a binary function");
233 static_assert(not std::is_pointer_v<F>, "functor must not be function pointer");
234
235 static constexpr bool is_commutative = op_traits<functor>::is_commutative;
236 static std::unique_ptr<functor> f;
237
238 static void apply(void *invec, void *inoutvec, int *len, MPI_Datatype *datatype) {
239 T *i1 = static_cast<T *>(invec);
240 T *i2 = static_cast<T *>(inoutvec);
241 for (int i{0}, i_end{*len}; i < i_end; ++i, ++i1, ++i2)
242 *i2 = (*f)(*i1, *i2);
243 }
244
245 MPI_Op mpi_op{MPI_OP_NULL};
246
247 private:
248 explicit op(F f_) {
249 f.reset(new F(f_));
250 MPI_Op_create(op::apply, is_commutative, &mpi_op);
251 }
252
253 public:
254 op(op const &) = delete;
255
256 ~op() { MPI_Op_free(&mpi_op); }
257
258 void operator=(op const &) = delete;
259
260 friend op &get_op<>(F);
261 };

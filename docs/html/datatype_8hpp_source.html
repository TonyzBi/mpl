139  template<typename T>
140  struct_layout &register_element(T &x) {
141  static_assert(not std::is_const<T>::value, "type must not be const");
142  static_assert(not std::is_pointer<T>::value, "type must not be pointer");
143  blocklengths.push_back(size(x));
144  MPI_Aint address;
145  MPI_Get_address(&x, &address);
146  displacements.push_back(address - base);
147  datatypes.push_back(get_datatype(x));

160  template<typename T>
161  class base_struct_builder {
162  private:
163  MPI_Datatype type;
164 
165  protected:
166  void define_struct(const struct_layout<T> &str) {
167  MPI_Datatype temp_type;
168  MPI_Type_create_struct(str.blocklengths.size(), str.blocklengths.data(),
169  str.displacements.data(), str.datatypes.data(), &temp_type);
170  MPI_Type_commit(&temp_type);
171  MPI_Type_create_resized(temp_type, 0, sizeof(T), &type);
172  MPI_Type_commit(&type);
173  MPI_Type_free(&temp_type);
174  }
175 
176  base_struct_builder() = default;
177 
178  base_struct_builder(const base_struct_builder &) = delete;
179 
180  void operator=(const base_struct_builder &) = delete;
181 
182  ~base_struct_builder() { MPI_Type_free(&type); }
183 
184  using data_type_category = detail::basic_or_fixed_size_type;
185 
186  friend class detail::datatype_traits_impl<T, void>;
187  };

235 
236  void operator()(T &x) const {

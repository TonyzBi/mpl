59 template<typename S>
60 class struct_layout {
61 template<typename T>
62 inline constexpr std::size_t size(const T &) const {
63 return 1;
64 }
65
66 template<typename T, std::size_t N0>
67 inline constexpr std::size_t size(const std::array<T, N0> &) const {
68 return N0;
69 }
70
71 template<typename T, std::size_t N0>
72 inline constexpr std::size_t size(const T (&)[N0]) const {
73 return N0;
74 }
75
76 template<typename T, std::size_t N0, std::size_t N1>
77 inline constexpr std::size_t size(const T (&)[N0][N1]) const {
78 return N0 * N1;
79 }
80
81 template<typename T, std::size_t N0, std::size_t N1, std::size_t N2>
82 inline constexpr std::size_t size(const T (&)[N0][N1][N2]) const {
83 return N0 * N1 * N2;
84 }
85
86 template<typename T, std::size_t N0, std::size_t N1, std::size_t N2, std::size_t N3>
87 inline constexpr std::size_t size(const T (&)[N0][N1][N2][N3]) const {
88 return N0 * N1 * N2 * N3;
89 }
90
91 template<typename T>
92 inline MPI_Datatype get_datatype(const T &) const {
93 return detail::datatype_traits<T>().get_datatype();
94 }
95
96 template<typename T, std::size_t N0>
97 inline MPI_Datatype get_datatype(const std::array<T, N0> &) const {
98 return detail::datatype_traits<T>().get_datatype();
99 }
100
101 template<typename T, std::size_t N0>
102 inline MPI_Datatype get_datatype(const T (&)[N0]) const {
103 return detail::datatype_traits<T>().get_datatype();
104 }
105
106 template<typename T, std::size_t N0, std::size_t N1>
107 inline MPI_Datatype get_datatype(const T (&)[N0][N1]) const {
108 return detail::datatype_traits<T>().get_datatype();
109 }
110
111 template<typename T, std::size_t N0, std::size_t N1, std::size_t N2>
112 inline MPI_Datatype get_datatype(const T (&)[N0][N1][N2]) const {
113 return detail::datatype_traits<T>().get_datatype();
114 }
115
116 template<typename T, std::size_t N0, std::size_t N1, std::size_t N2, std::size_t N3>
117 inline MPI_Datatype get_datatype(const T (&)[N0][N1][N2][N3]) const {
118 return detail::datatype_traits<T>().get_datatype();
119 }
120
121 MPI_Aint base_{};
122 std::vector<int> block_lengths_;
123 std::vector<MPI_Aint> displacements_;
124 std::vector<MPI_Datatype> data_types_;
125
126 public:
130 struct_layout &register_struct(const S &x) {
131 MPI_Get_address(const_cast<S *>(&x), &base_);
132 return *this;
133 }
134
139 template<typename T>
140 struct_layout &register_element(T &x) {
141 static_assert(not std::is_const<T>::value, "type must not be const");
142 static_assert(not std::is_pointer<T>::value, "type must not be pointer");
143 block_lengths_.push_back(size(x));
144 MPI_Aint address;
145 MPI_Get_address(&x, &address);
146 displacements_.push_back(address - base_);
147 data_types_.push_back(get_datatype(x));
148 return *this;
149 }
150
151 friend class base_struct_builder<S>;
152 };

160 template<typename T>
161 class base_struct_builder {
162 private:
163 MPI_Datatype type_;
164
165 protected:
166 void define_struct(const struct_layout<T> &str) {
167 MPI_Datatype temp_type;
168 MPI_Type_create_struct(str.block_lengths_.size(), str.block_lengths_.data(),
169 str.displacements_.data(), str.data_types_.data(), &temp_type);
170 MPI_Type_commit(&temp_type);
171 MPI_Type_create_resized(temp_type, 0, sizeof(T), &type_);
172 MPI_Type_commit(&type_);
173 MPI_Type_free(&temp_type);
174 }
175
176 base_struct_builder() = default;
177
178 public:
179 base_struct_builder(const base_struct_builder &) = delete;
180 void operator=(const base_struct_builder &) = delete;
181
182 protected:
183 ~base_struct_builder() { MPI_Type_free(&type_); }
184
185 using data_type_category = detail::basic_or_fixed_size_type;
186
187 friend class detail::datatype_traits_impl<T, void>;
188 };

230 template<typename F, typename T, std::size_t n>
231 class apply_n {
232 F &f_;
233
234 public:
235 explicit apply_n(F &f) : f_{f} {}
236
237 void operator()(T &x) const {
238 apply_n<F, T, n - 1> next(f_);
239 next(x);
240 f_(std::get<n - 1>(x));
241 }
242 };
243
244 template<typename F, typename T>
245 class apply_n<F, T, 1> {
246 F &f_;
247
248 public:
249 explicit apply_n(F &f) : f_{f} {}
250
251 void operator()(T &x) const { f_(std::get<0>(x)); }
252 };
253
254 template<typename F, typename... Args>
255 void apply(std::tuple<Args...> &t, F &f) {
256 apply_n<F, std::tuple<Args...>, std::tuple_size<std::tuple<Args...>>::value> app(f);
257 app(t);
258 }

263
264 public:

266
267 template<typename T>
268 void operator()(T &x) const {
